Real-time Talent Analytics

Our system processes vast amounts of candidate data in real-time using Azure Stream Analytics:

WITH CandidateStream AS (
    SELECT 
        ApplicantId,
        Name,
        Skills,
        Experience,
        InteractionType,
        ResponseTime,
        EventEnqueuedUtcTime
    FROM 
        ApplicantInteractionStream TIMESTAMP BY EventEnqueuedUtcTime
)
SELECT 
    ApplicantId,
    Name,
    Skills,
    AVG(ResponseTime) OVER (PARTITION BY ApplicantId 
                            ORDER BY EventEnqueuedUtcTime 
                            ROWS BETWEEN 20 PRECEDING AND CURRENT ROW) as AvgResponseTime,
    COUNT(CASE WHEN InteractionType = 'Positive' THEN 1 END) OVER (PARTITION BY ApplicantId) as PositiveInteractions,
    PERCENTILE_CONT(0.5) WITHIN GROUP (ORDER BY Experience) OVER (PARTITION BY Skills) as MedianExperienceBySkill
INTO 
    PowerBIOutput
FROM 
    CandidateStream

const onboardEmployee = async (employeeData) => {
    try {
        const adpResponse = await fetch('https://api.adp.com/hr/v2/workers', {
            method: 'POST',
            headers: {
                'Authorization': `Bearer ${ADP_API_TOKEN}`,
                'Content-Type': 'application/json'
            },
            body: JSON.stringify(employeeData)
        });
        const adpData = await adpResponse.json();
        
        // Sync with internal systems
        await syncInternalSystems(adpData);
        
        // Trigger onboarding workflow
        const workflowResponse = await fetch('/api/internal/onboarding-workflow', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ employeeId: adpData.workerId })
        });
        const workflowData = await workflowResponse.json();
        
        console.log('Employee onboarded successfully:', { adp: adpData, workflow: workflowData });
    } catch (error) {
        console.error('Onboarding failed:', error);
        // Implement retry logic or manual intervention notification
    }
};

const syncInternalSystems = async (adpData) => {
    // Sync with HRIS, payroll, and access management systems
    const syncPromises = [
        syncHRIS(adpData),
        syncPayroll(adpData),
        provisionAccess(adpData)
    ];
    await Promise.all(syncPromises);
};
